# Consultar boliche segun posicion:
curl -v -X GET 'http://localhost:9090/boliches?lat=-31.337485&lon=-64.256344' | python -m json.tool



# Insertar un boliche:
curl -v -H "Content-Type: application/json" -X POST -d '{"nombre":"labs","latitud":"-31.337332","longitud":"-64.256521"}' 'http://localhost:9090/boliches' | python -m json.tool



# Obtener tema actual:
curl -v -H "X-LAT: -31.337485" -H "X-LON: -64.256521" -X GET 'http://localhost:9090/boliches/1/tema_actual' | python -m json.tool



# Insertar el tema actual en un boliche:
curl -v -H "Content-Type: application/json" -H "X-LAT: -31.337485" -H "X-LON: -64.256521" -X POST -d '{"nombre_tema":"criminal - ozuna"}'  'http://localhost:9090/boliches/1/tema_actual' | python -m json.tool
